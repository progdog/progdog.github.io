import pandas as pd

d = {
    "name": ["Kevin", "Jack", "Mary"],
    "sex": ["Male", "Male", "Female"],
    "score": [80, 90, 95]
}

df = pd.DataFrame(data=d)
name sex score
Kevin Male 80
Jack Male 90
Mary Female 95

Single condition filter

result = df.loc[df['score']>=90]
# OR
result = df[df['score']>=90]

Result

name sex score
Jack Male 90
Mary Female 95

Multiple conditions filter

result = df[(df['score'] >= 90) & (df['sex'] == 'Female')]

Result

name sex score
Mary Female 95
